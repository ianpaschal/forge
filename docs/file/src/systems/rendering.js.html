    const Three = require( "three" );
const OrbitControls = require( "three-orbit-controls" )( Three );

export default class {
	constructor() {
		console.log( "Created a new rendering system." );
		this._mixers = [];
		this.init();
	}
	init() {
		console.log( "Initializing a new rendering system." );

		const container = document.createElement( "div" );
		console.log( document );
		document.body.appendChild( container );
		this._camera = new Three.PerspectiveCamera( 30, window.innerWidth / window.innerHeight, 1, 10000 );
		this._camera.up.set( 0, 0, 1 );
		this._camera.position.set( 8, -8, 8 );
		this._camera.lookAt( new Three.Vector3( 0, 0, 0 ) );

		const loader = new Three.JSONLoader();

		const scope = this;

		const paths = [];
		paths.push( "../../plugins/age-of-mythology/model/greek-villager-female-forage.js" );
		paths.forEach( ( path ) => {
			loader.load( path, ( geometry, materials ) => {
				for ( var i = 0; i < materials.length; i++ ) {
					materials[ i ].color.setRGB( 1.0, 1.0, 1.0 );
					materials[ i ].morphTargets = true;
				}
				const mesh = new Three.Mesh( geometry, new Three.MeshLambertMaterial({
					color: 0x999999
				}) );
				mesh.material.morphTargets = true;

				const mixer = new Three.AnimationMixer( mesh );
				const clip = Three.AnimationClip.CreateFromMorphTargetSequence( "walk", geometry.morphTargets, 30 );
				mixer.clipAction( clip ).setDuration( 1 ).play();

				scope._scene.add( mesh );
				mesh.position.z -= 1;
				scope._mixers.push( mixer );
			});
		});

		this._scene = new Three.Scene();

		let light;
		light = new Three.DirectionalLight( 0xefefff, 1 );
		light.position.set( 1, 1, 1 ).normalize();
		this._scene.add( light );
		light = new Three.DirectionalLight( 0xefccaa, 1 );
		light.position.set( -1, -1, -1 ).normalize();
		this._scene.add( light );
		this._scene.add( new Three.AmbientLight( 0x666666 ) );

		this._renderer = new Three.WebGLRenderer();
		this._renderer.setClearColor( 0x999999 );
		this._renderer.setPixelRatio( window.devicePixelRatio );
		this._renderer.setSize( window.innerWidth, window.innerHeight );
		this._renderer.antialias = false;

		container.appendChild( this._renderer.domElement );
		window.addEventListener( "resize", this._resize.bind( this ), false );

		this._controls = new OrbitControls( this._camera, this._renderer.domElement );
		this._controls.enableRotate = true;
	}
	update( delta ) {
		this._mixers.forEach( ( mixer ) => {
			mixer.update( delta / 1000 );
		});
		this._renderer.render( this._scene, this._camera );
	}
	_resize( e ) {
		this._camera.aspect = window.innerWidth / window.innerHeight;
		this._camera.updateProjectionMatrix();
		this._renderer.setSize( window.innerWidth, window.innerHeight );
	}
}
